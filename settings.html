export default { async fetch(request) { const url = new URL(request.url); const target = url.searchParams.get('url'); if (!target) return new Response('url required', { status: 400 }); const res = await fetch(decodeURIComponent(target), { headers: { 'User-Agent': 'Mozilla/5.0' } }); return new Response(await res.text(), { headers: { 'Content-Type': 'text/html; charset=utf-8', 'Access-Control-Allow-Origin': '*', } }); } }
